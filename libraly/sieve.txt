エラトステネスのふるい O(n)
----------------------------------------------


## エラトステネスのふるい
vector<bool> primes;
void make_primes(int n)
{
	primes.resize(n+1, true);
	primes[0] = primes[1] = false;
	rep(i, 2, sqrt(n))
	{
		if (primes[i])
		{
			for (int j = 0; i * (j + 2) < n; j++)
				primes[i * (j + 2)] = false;
		}
	}
}
